## Test2

Testing R-R forwarding and the functionality of lw.

```C
//D[0] = 10;
//D[1] = 20;
//D[2] = 30;
int a = D[0];
int b = D[1];
int c = D[2];
d = a - b; //d = -10
c = a + b; //c = 30
a = c + a; //a = 40
a = a + a; //a = 80
c = a + a; //c = 160
D[3] = c; //D[3] = 160
D[4] = d; //D[4] = -10

```

Assembly:

```python
lw $1 , 0 //int a = D[0];
lw $2 , 1 //int b = D[1];
lw $3 , 2 //int c = D[2];
nop // To prevent lw-r hazard first
nop
sub $4,$1,$2 // d = a - b
add $3,$1,$2 // c = a + b
add $1,$3,$1 // a = c + a , forwarding $3
add $1,$1,$1 // a = a + a , forwarding $1
add $3,$1,$1 // c = a + a , forwarding $1
sw  $3 , 3   // D[3]      , forwarding $3
sw  $4 , 4   // D[4]
nop
nop
nop
nop
stop
//Check D[3] = 160 and D[4] = -10

```
